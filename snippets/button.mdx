export const Button = ({
  label,
  href,
}) => {
  const handleClick = () => {
    if (href) {
      if (href.startsWith('http')) {
        window.open(href, '_blank');
      } else {
        window.location.href = href;
      }
    }
  };

  return (
    <button 
      className="bg-blue-600 hover:bg-blue-700 text-white px-4 py-2 rounded-md font-medium transition-colors duration-200"
      onClick={handleClick}
    >
      {label}
    </button>
  );
}
